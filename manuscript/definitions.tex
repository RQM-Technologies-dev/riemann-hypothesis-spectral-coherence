\begin{definition}[Completed zeta function]
\[
\xi(s) := \tfrac12 s(s-1)\pi^{-s/2}\Gamma(s/2)\zeta(s).
\]
We use the standard symmetry $\xi(s)=\xi(1-s)$.
\end{definition}

\begin{definition}[Critical strip and critical line]
The critical strip is $\{s\in\C:0<\RePart(s)<1\}$. The critical line is $\{s\in\C:\RePart(s)=\tfrac12\}$.
\end{definition}

\begin{definition}[Reflection involution]
Let $\sigma:\C\to\C$ be $\sigma(s)=1-s$. The fixed-point set of $\sigma$ is the critical line.
\end{definition}
