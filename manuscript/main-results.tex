\subsection*{Established inputs (from standard references)}
\begin{proposition}[Functional symmetry]
\label{prop:functional-symmetry}
The completed zeta function
\[
\xi(s)=\tfrac12 s(s-1)\pi^{-s/2}\Gamma\!\left(\tfrac{s}{2}\right)\zeta(s)
\]
satisfies $\xi(s)=\xi(1-s)$; see \cite{Riemann1859,Edwards1974,Titchmarsh1986}.
\end{proposition}

\begin{proposition}[Zero-set symmetries]
\label{prop:zero-symmetries}
If $\rho$ is a nontrivial zero of $\zeta$, then $1-\rho$ and $\overline{\rho}$ are also nontrivial zeros; see \cite{Edwards1974,Titchmarsh1986}.
\end{proposition}
