\subsection*{Established Results (from the Literature)}

The following results are standard and are cited without proof.

\begin{proposition}[Functional Equation, \cite{Riemann1859, Titchmarsh1986}]
  \label{prop:functional-equation}
  The completed zeta function $\xi(s) = \frac{1}{2}s(s-1)\pi^{-s/2}\Gamma(s/2)\zeta(s)$
  satisfies $\xi(s) = \xi(1-s)$ for all $s \in \C$.
\end{proposition}

\begin{proposition}[Symmetry of Zeros, \cite{Titchmarsh1986}]
  \label{prop:zero-symmetry}
  The nontrivial zeros of $\zeta$ are symmetric under $s \mapsto 1-s$ and
  under $s \mapsto \bar{s}$. In particular, they occur in quadruples
  $\{s, 1-s, \bar{s}, 1-\bar{s}\}$ unless they lie on the critical line
  or the real axis.
\end{proposition}

\begin{conjecture}[Main Claim — Riemann Hypothesis]
  \label{conj:main}
  All nontrivial zeros of $\zeta(s)$ lie on the critical line $\Re(s) = \tfrac{1}{2}$.
\end{conjecture}

  \label{conj:correspondence}

  $s \in \C$ in the critical strip is a nontrivial zero of $\zeta$ if and only if
  $s$ satisfies the balance criterion (Definition~\ref{def:balance}).
\end{conjecture}

\subsection*{Auxiliary Results (proposed)}

\begin{conjecture}[Symmetry of the Balance Criterion]
  \label{conj:symmetry}
  The balance criterion is symmetric under the reflection $\sigma \colon s \mapsto 1-s$.
  That is, $s$ satisfies the balance criterion if and only if $\sigma(s) = 1-s$ does.
\end{conjecture}
