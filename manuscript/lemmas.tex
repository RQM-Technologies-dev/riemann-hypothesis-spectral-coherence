\subsection*{A. Classical symmetry inputs (available; standard references)}

\begin{lemma}[Reflection transport of nontrivial zeros]
If $\rho$ is a nontrivial zero of $\zeta$, then $1-\rho$ is also a nontrivial zero.
\end{lemma}
\noindent\emph{Status: standard.} This is a direct consequence of the completed zeta function functional equation $\xi(s)=\xi(1-s)$; see \cite{Edwards1974,Titchmarsh1986}.

\begin{lemma}[Conjugation transport of nontrivial zeros]
If $\rho$ is a nontrivial zero of $\zeta$, then $\overline{\rho}$ is also a nontrivial zero.
\end{lemma}
\noindent\emph{Status: standard.} This follows from Schwarz reflection for $\xi$ (equivalently $\xi(\bar s)=\overline{\xi(s)}$); see \cite{Edwards1974,Titchmarsh1986}.
